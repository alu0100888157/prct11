def to_s
    s = @titulo + " (" + @porcentaje + "%" + ")" + "\n"
    i=0;
    for i in 0..@descripcion.size-1
        s += "- " + "#{@descripcion[i]}" + ": " + "#{@porcion[i]}" + ", " + "#{@gramos[i]}g\n"
    end
    
    s+= "V.C.T | %    " + @vct + "kcal | " + @p_proteina + "% - " + @p_grasas + "% - " + @p_hidratos + "%"
    s
end
